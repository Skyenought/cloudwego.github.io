Hertz 提供了国际化 (i18n) 的[中间件扩展](https://github.com/hertz-contrib/i18n) ，它参考了 Gin 的[实现](https://github.com/gin-contrib/i18n) 。

使用方法可参考如下 [example](https://github.com/hertz-contrib/i18n/tree/main/example)

### 方法
```go
// 用于将 i18n 扩展集成进 hertz server
func Localize(opts ...Option) app.HandlerFunc
```

### 配置函数
```go
// 用于将自定义配置加载进入中间件
func WithBundle(cfg *BundleCfg) Option

// 用于配置 i18n 模板触发条件, 可以通过从参数，请求头中取出信息
func WithGetLangHandle(handler GetLangHandler)
```

### 配置详情
```go
type (
    BundleCfg struct {
        // 默认转换语言类型
        DefaultLanguage  language.Tag
        // 转换文件模板类型，例如: yaml, json
        FormatBundleFile string
        // 接收转换类型
        AcceptLanguage   []language.Tag
        // 模板文件目录
        RootPath         string
        // 模板文件解码函数，例如: yaml.Unmarshal
        UnmarshalFunc    i18n.UnmarshalFunc
        // 文件读取函数, 例如 LoaderFunc(ioutil.ReadFile)
        Loader           Loader
    }

    Loader interface {
        LoadMessage(path string) ([]byte, error)
    }

    GetLangHandler = func(c context.Context, ctx *app.RequestContext, defaultLang string) string
)

```

### 默认配置

```go
BundleCfg {
	// 默认模板路径，必须自行进行指定
	RootPath:         defaultRootPath,
	AcceptLanguage:   []language.Tag{
		defaultLanguage,
		language.Chinese,
	},
	FormatBundleFile: "yaml",
	DefaultLanguage:  language.English,
	UnmarshalFunc:    yaml.Unmarshal,
	// 默认使用 io.ReadFile 来读取文件
	Loader:           LoaderFunc(ioutil.ReadFile),
}
```

### 简略使用案例

```go
package main

import (
	"context"

	"github.com/cloudwego/hertz/pkg/app"
	"github.com/cloudwego/hertz/pkg/app/server"
	hertzI18n "github.com/hertz-contrib/i18n"
	"github.com/nicksnyder/go-i18n/v2/i18n"
	"golang.org/x/text/language"
	"gopkg.in/yaml.v3"
)

func main() {
	h := server.New(server.WithHostPorts(":3000"))
	h.Use(hertzI18n.Localize(
		hertzI18n.WithBundle(&hertzI18n.BundleCfg{
			RootPath:         "./localize",
			AcceptLanguage:   []language.Tag{language.Chinese, language.English},
			DefaultLanguage:  language.Chinese,
			FormatBundleFile: "yaml",
			UnmarshalFunc:    yaml.Unmarshal,
		}),
		hertzI18n.WithGetLangHandle(func(c context.Context, ctx *app.RequestContext, defaultLang string) string {
			// 配置从 lang 参数中取出语言类型，为空使用默认语言类型
			lang := ctx.Query("lang")
			if lang == "" {
				return defaultLang
			}
			return lang
		}),
	))
	// 例如当请求地址为: http://127.0.0.1:3000/user 时, 响应结果为 你好 user
	// 当请求地址为: http://127.0.0.1:3000/user?lang=en 时, 响应结果为 welcome user
	h.GET("/:name", func(c context.Context, ctx *app.RequestContext) {
		ctx.String(200, hertzI18n.MustGetMessage(&i18n.LocalizeConfig{
			MessageID: "welcomeWithName",
			TemplateData: map[string]string{
				"name": ctx.Param("name"),
			},
		}))
	})
	// 例如当请求地址为: http://127.0.0.1:3000/ 时, 响应结果为 你好
	// 当请求地址为: http://127.0.0.1:3000/?lang=en 时, 响应结果为 welcome
	h.GET("/", func(c context.Context, ctx *app.RequestContext) {
		ctx.String(200, hertzI18n.MustGetMessage("welcome"))
	})

	h.Spin()
}
```

完整用法示例详见 [i18n](https://github.com/hertz-contrib/i18n/)
